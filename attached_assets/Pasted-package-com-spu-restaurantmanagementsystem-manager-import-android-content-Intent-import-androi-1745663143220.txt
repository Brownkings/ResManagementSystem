import androidx.appcompat.app.AppCompatActivity;

import com.google.firebase.auth.FirebaseAuth;
import com.google.firebase.database.DataSnapshot;
import com.google.firebase.database.DatabaseError;
import com.google.firebase.database.ValueEventListener;
import com.spu.restaurantmanagementsystem.LoginActivity;
import com.spu.restaurantmanagementsystem.R;
import com.spu.restaurantmanagementsystem.models.User;
import com.spu.restaurantmanagementsystem.utils.FirebaseUtil;

public class ManagerDashboardActivity extends AppCompatActivity {

    private TextView textViewWelcome;
    private CardView cardViewStaff, cardViewMenu, cardViewInventory, cardViewReservation, cardViewReports;
    private Button buttonLogout;
    private FirebaseAuth mAuth;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_manager_dashboard);

        // Initialize UI elements
        textViewWelcome = findViewById(R.id.textViewWelcome);
        cardViewStaff = findViewById(R.id.cardViewStaff);
        cardViewMenu = findViewById(R.id.cardViewMenu);
        cardViewInventory = findViewById(R.id.cardViewInventory);
        cardViewReservation = findViewById(R.id.cardViewReservation);
        cardViewReports = findViewById(R.id.cardViewReports);
        buttonLogout = findViewById(R.id.buttonLogout);

        // Initialize Firebase Auth
        mAuth = FirebaseAuth.getInstance();

        // Load user data
        loadUserData();

        // Set up click listeners
        cardViewStaff.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Navigate to staff management activity
                // startActivity(new Intent(ManagerDashboardActivity.this, StaffManagementActivity.class));
            }
        });

        cardViewMenu.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Navigate to menu management activity
                // startActivity(new Intent(ManagerDashboardActivity.this, MenuManagementActivity.class));
            }
        });

        cardViewInventory.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Navigate to inventory activity
                // startActivity(new Intent(ManagerDashboardActivity.this, InventoryActivity.class));
            }
        });

        cardViewReservation.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Navigate to reservation management activity
                // startActivity(new Intent(ManagerDashboardActivity.this, ReservationManagementActivity.class));
            }
        });

        cardViewReports.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Navigate to reports activity
                // startActivity(new Intent(ManagerDashboardActivity.this, ReportsActivity.class));
            }
        });

        buttonLogout.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Sign out
                mAuth.signOut();

                // Navigate to login activity
                startActivity(new Intent(ManagerDashboardActivity.this, LoginActivity.class));
                finish();
            }
        });
    }

    private void loadUserData() {
        String userId = FirebaseUtil.getCurrentUser().getUid();

        FirebaseUtil.getUsersRef().child(userId).addListenerForSingleValueEvent(new ValueEventListener() {
            @Override
            public void onDataChange(@NonNull DataSnapshot snapshot) {
                if (snapshot.exists()) {
                    User user = snapshot.getValue(User.class);
                    if (user != null) {
                        textViewWelcome.setText("Welcome, " + user.getName());
                    }
                }
            }

            @Override
            public void onCancelled(@NonNull DatabaseError error) {
                // Handle error
            }
        });
    }
}
